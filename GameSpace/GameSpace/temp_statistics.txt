        /// <summary>
        /// 每日遊戲次數限制統計頁面
        /// </summary>
        public async Task<IActionResult> Statistics()
        {
            try
            {
                var statistics = await _service.GetStatisticsAsync();
                var viewModel = new DailyGameLimitStatisticsViewModel
                {
                    TotalSettings = statistics.TotalSettings,
                    EnabledSettings = statistics.EnabledSettings,
                    DisabledSettings = statistics.DisabledSettings,
                    AverageDailyLimit = statistics.AverageDailyLimit,
                    MinDailyLimit = statistics.MinDailyLimit,
                    MaxDailyLimit = statistics.MaxDailyLimit,
                    LastUpdated = statistics.LastUpdated
                };

                _logger.LogInformation("顯示每日遊戲次數限制統計資料");
                return View(viewModel);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "顯示每日遊戲次數限制統計資料時發生錯誤");
                TempData["ErrorMessage"] = "載入統計資料時發生錯誤，請稍後再試";
                return RedirectToAction(nameof(Index));
            }
        }
